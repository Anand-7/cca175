#Import all tables from MySQL database retail_db into Hive default database
sqoop import-all-tables \
--connect jdbc:mysql://quickstart.cloudera:3306/retail_db \
--username retail_dba \
--password cloudera \
--fields-terminated-by ',' \
--hive-import \
--create-hive-table \
--hive-overwrite
